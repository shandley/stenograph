import pandas as pd

def fdr_correction(pvals):
    """Benjamini-Hochberg FDR correction"""
    n = len(pvals)
    ranked = np.argsort(pvals)
    adjusted = np.zeros(n)
    for i, idx in enumerate(reversed(ranked)):
        rank = n - i
        if i == 0:
            adjusted[idx] = pvals[idx]
        else:
            adjusted[idx] = min(pvals[idx] * n / rank, adjusted[ranked[n - i]])
    return np.minimum(adjusted, 1.0)
