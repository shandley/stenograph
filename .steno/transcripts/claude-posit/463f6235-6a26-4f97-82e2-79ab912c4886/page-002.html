claude_cli_query <- function(prompt, context = NULL, working_dir = getwd()) {
  if (!claude_cli_available()) {
    stop(
      "Claude CLI not found. Install from https://code.claude.com\n",
      "Or use backend = 'api' instead.",
      call. = FALSE
    )
  }

  # Create temporary input file
  input_file <- tempfile(fileext = ".txt")
  on.exit(unlink(input_file), add = TRUE)

  # Build prompt with context
  full_prompt <- build_cli_prompt(prompt, context)
  writeLines(full_prompt, input_file)

  # Execute CLI
  output <- tryCatch({
    system2(
      "claude",
      args = c("--input", shQuote(input_file)),
      stdout = TRUE,
      stderr = TRUE,
      env = c(
        paste0("HOME=", Sys.getenv("HOME")),
        paste0("PATH=", Sys.getenv("PATH"))
      )
    )
  }, error = function(e) {
    stop("Claude CLI execution failed: ", conditionMessage(e), call. = FALSE)
  })

  # Check for errors in output
  if (!is.null(attr(output, "status")) && attr(output, "status") != 0) {
    stop("Claude CLI returned error: ", paste(output, collapse = "\n"), call. = FALSE)
  }

  paste(output, collapse = "\n")
}

claude_cli_query <- function(prompt, context = NULL, working_dir = getwd()) {
  if (!claude_cli_available()) {
    stop(
      "Claude CLI not found. Install from https://code.claude.com\n",
      "Or use backend = 'api' instead.",
      call. = FALSE
    )
  }

  # Build prompt with context
  full_prompt <- build_cli_prompt(prompt, context)

  # Claude Code CLI reads from stdin, not --input flag
  # We'll use pipe to send the prompt
  output <- tryCatch({
    # Write prompt to temp file for piping
    temp_input <- tempfile(fileext = ".txt")
    on.exit(unlink(temp_input), add = TRUE)
    writeLines(full_prompt, temp_input)

    # Pipe to claude CLI
    cmd <- sprintf("cat %s | claude", shQuote(temp_input))
    system(cmd, intern = TRUE)
  }, error = function(e) {
    # If that fails, try direct stdin (for interactive mode)
    tryCatch({
      system2(
        "claude",
        args = character(0),
        stdout = TRUE,
        stderr = TRUE,
        input = full_prompt
      )
    }, error = function(e2) {
      stop(
        "Claude CLI execution failed. ",
        "The CLI may not support non-interactive mode.\n",
        "Consider using backend = 'api' instead.",
        call. = FALSE
      )
    })
  })

  if (length(output) == 0) {
    stop("Claude CLI returned no output", call. = FALSE)
  }

  paste(output, collapse = "\n")
}
